name: ci

on:
  push:
    branches:
      - master
      - main

permissions:
  contents: read  # Only reading repository content, not pushing to GitHub Pages

jobs:
  publish_to_hashnode:
    runs-on: ubuntu-latest

    steps:
      - name: Checkout code
        uses: actions/checkout@v4

      - name: Set up Python
        uses: actions/setup-python@v5
        with:
          python-version: 3.x

      - name: Install dependencies
        run: pip install mkdocs-material requests  # 'requests' used in the Python script

      - name: Build MkDocs site
        run: mkdocs build --clean  # This generates the static site in the 'site/' directory

      - name: Publish to Hashnode
        env:
          HASHNODE_API_TOKEN: ${{ secrets.HASHNODE_API_TOKEN }}  # Add this secret in GitHub repo settings
          HASHNODE_BLOG_ID: ${{ secrets.HASHNODE_BLOG_ID }}  # Replace with your actual Hashnode blog ID
        run: python scripts/publish_to_hashnode.py  # Ensure this script exists and handles upload to Hashnode
